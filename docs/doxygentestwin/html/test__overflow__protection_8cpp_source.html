190TEST_F(OverflowProtectionTest, TestOverflowPreventionAlgorithm) {
191 // Complete overflow prevention algorithm
192 auto overflowSafeCalculation = [](long long counter, long long frequency, long long periodDen) -> long long {
193 // Check if direct multiplication would overflow
194 if (counter > 0 && periodDen > 0) {
195 if (counter > std::numeric_limits<long long>::max() / periodDen) {
196 // Use safe algorithm
197 const long long whole = (counter / frequency) * periodDen;
198 const long long part = (counter % frequency) * periodDen / frequency;
199
200 // Check if result would overflow
201 if (whole > std::numeric_limits<long long>::max() - part) {
202 throw std::overflow_error("Calculation would overflow even with safe algorithm");
203 }
204
205 return whole + part;
206 }
207 }
208
209 // Safe to use direct multiplication
210 return (counter * periodDen) / frequency;
211 };
